// 2. Pass a function to another function

package main

import "fmt"

func doSomething(name string, action func(name string)) {
// that function is being called in here
action(name)
}

func doing(name string) {
fmt.Println("Hello", name)
}

func main() {

    // doSomething("Ram", doing)

    doSomething("Ram", func(name string) {

    	// this is only passing the reference of the function
    	fmt.Println("Hello", name)

    })

}
