// 3. Store functions in a map
package main

import "fmt"

func main() {

	actions := map[string]func(string){
		"greet": func(name string) { fmt.Println("Hello", name) },
		"shout": func(name string) { fmt.Println("HELLO", name) },
	}

	actions["greet"]("Ram")

	simpleMap := map[string]string{
		"name": "Ronaldo",
	}

	fmt.Println(simpleMap["name"])
}
