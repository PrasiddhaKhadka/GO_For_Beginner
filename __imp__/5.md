// Step 3: Higher-Order Functions

// A higher-order function is a function that either:

// Takes a function as input, OR
// Returns a function as output, OR
// Both

package main

import "fmt"

type AppHandler func(name string) error

func run(name string, handler AppHandler) {
	err := handler(name)

	if err != nil {
		fmt.Println("error: ", err)
	}

}

func main() {

	appHandler := AppHandler(func(name string) error {
		fmt.Println("Hello", name)
		return nil
	})

	run("Ram", appHandler)
}
