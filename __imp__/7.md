package main

import (
	"errors"
	"fmt"
	"time"
)

type NameHandler func(name string) error

func withLogger(next NameHandler) NameHandler {

	return func(name string) error {
		fmt.Println("[LOG] before — name:", name)
		err := next(name)
		fmt.Println("[LOG] after  — name:", name)
		return err
	}
}

// wrapper 2 — adds timing
func withTimer(next NameHandler) NameHandler {
	return func(name string) error {
		start := time.Now()           // ← record start time
		err := next(name)             // ← call the original
		duration := time.Since(start) // ← calculate how long it took
		fmt.Printf("[TIMER] took: %v\n", duration)
		return err
	}
}

func main() {

	greet := NameHandler(func(name string) error {
		fmt.Println("Hello", name)
		time.Sleep(100 * time.Millisecond)
		return nil
	})

	fail := NameHandler(func(name string) error {
		return errors.New("something went wrong")
	})

	loggedGreet := withLogger(greet)

	// calling original — no logging
	fmt.Println("=== without logger ===")
	greet("Ram")

	// calling wrapped — logging appears
	fmt.Println("=== with logger ===")
	loggedGreet("Ram")

	wrappedGreet := withLogger(withTimer(greet))
	wrappedFail := withLogger(withTimer(fail))

	fmt.Println("=== greet ===")
	wrappedGreet("Ram")

	fmt.Println()
	fmt.Println("=== fail ===")
	wrappedFail("Ram")
}
