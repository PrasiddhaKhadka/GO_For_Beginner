// Step 4: Closures

// A closure is a function that remembers the variables from the scope it was created in —
//  even after that scope has finished executing.

package main

import "fmt"

func main() {
	message := "Hello Ram" // ← lives in main's scope

	say := func() {
		fmt.Println(message) // ← inner function uses message from OUTSIDE itself
	}

	say() // Hello Ram
}
