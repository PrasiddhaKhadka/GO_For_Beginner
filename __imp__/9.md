package main

import "fmt"

func makeGreeter(greeting string, counter int) func(name string) {
// makeGreeter will FINISH executing and return
// but watch what happens to `greeting`

    return func(name string) {

    	fmt.Println(greeting, name) // ← greeting is from makeGreeter's scope
    	//  which is already DONE by the time this runs

    	counter += 1

    	fmt.Println(counter)
    	// return counter

    }

}

func main() {

    hello := makeGreeter("Hello", 0)
    hello("Ram")
    hello("Ram")
    hello("Ram")
    hello("Ram")

    hi := makeGreeter("Hi", 0)
    hi("Ram")
    hi("Ram")
    hi("Ram")
    hi("Ram")

}
